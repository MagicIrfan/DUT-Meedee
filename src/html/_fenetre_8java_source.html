235  public void trier(String attribut, boolean croissant)
236  {
237  if (tabalbums.isEmpty())
238  {
239  JOptionPane.showMessageDialog(this,"Impossible de trier");
240  }
241  else
242  {
243  panel_tableau.removeAll();
244  if (croissant)
245  {
246  if(attribut.equals("Titre"))
247  tabalbums.sort(Comparator.comparing(a -> a.getDonnees().getTitre(),String.CASE_INSENSITIVE_ORDER));
248  else if(attribut.equals("Auteur"))
249  tabalbums.sort(Comparator.comparing(a -> a.getDonnees().getAuteur(),String.CASE_INSENSITIVE_ORDER));
250  else if(attribut.equals("Date"))
251  tabalbums.sort(Comparator.comparing(a -> a.getDonnees().getDate()));
252  else if(attribut.equals("Genre"))
253  tabalbums.sort(Comparator.comparing(a -> a.getDonnees().getGenre(),String.CASE_INSENSITIVE_ORDER));
254  }
255  else
256  {
257  if(attribut.equals("Titre"))
258  {
259  tabalbums.sort(Comparator.comparing(a -> a.getDonnees().getTitre(), String.CASE_INSENSITIVE_ORDER));
260  Collections.reverse(tabalbums);
261  }
262  else if(attribut.equals("Auteur"))
263  {
264  tabalbums.sort(Comparator.comparing(a -> a.getDonnees().getAuteur(), String.CASE_INSENSITIVE_ORDER));
265  Collections.reverse(tabalbums);
266  }
267  else if(attribut.equals("Date"))
268  {
269  tabalbums.sort(Comparator.comparing(a -> a.getDonnees().getDate()));
270  Collections.reverse(tabalbums);
271  }
272  else if(attribut.equals("Genre"))
273  {
274  tabalbums.sort(Comparator.comparing(a -> a.getDonnees().getGenre(), String.CASE_INSENSITIVE_ORDER));
275  Collections.reverse(tabalbums);
276  }
277  }
278  for (int i=0; i<tabalbums.size();++i)
279  {
280  panel_tableau.add(tabalbums.elementAt(i));
281  }
282  panel_tableau.revalidate();
283  panel_tableau.repaint();
284  }
285  }
291  public void rechercher(String recherche)
292  {
293  if (b_filtre.getSelectedItem().toString().equals("Date"))
294  {
295  rempliralbumfiltredate(recherche,b_filtre_date.getSelectedItem().toString());
296  gestionnaire.rechercherAlbumDate(tabalbums,l_pseudo.getText(),recherche,b_filtre_date.getSelectedItem().toString());
297  }
298  else
299  {
300  rempliralbumfiltre(b_filtre.getSelectedItem().toString(),recherche);
301  gestionnaire.rechercherAlbum(tabalbums,l_pseudo.getText(),b_filtre.getSelectedItem().toString(),recherche);
302  }
303  if (!tabalbums.isEmpty())
304  b_reinitialiser.setEnabled(true);
305  panel_tableau.revalidate();
306  panel_tableau.repaint();
307  }
