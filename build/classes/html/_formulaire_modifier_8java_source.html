234  @Override
235  public boolean champsvalides()
236  {
237  boolean valide = true;
238  if (c_titre.isSelected())
239  {
240  if (t_titre.getText().isEmpty())
241  {
242  valide = false;
243  t_titre.setBackground(Color.red);
244  }
245  }
246  if (c_auteur.isSelected())
247  {
248  if (t_auteur.getText().isEmpty())
249  {
250  valide = false;
251  t_auteur.setBackground(Color.red);
252  }
253  }
254  if (c_date.isSelected())
255  {
256  if (t_date.getText().isEmpty())
257  {
258  valide = false;
259  t_date.setBackground(Color.red);
260  }
261  else
262  {
263  try
264  {
265  if ((Integer.parseInt(t_date.getText()) > Integer.parseInt(this.dateactuelle) || (Integer.parseInt(t_date.getText()) < 0)))
266  {
267  valide = false;
268  t_date.setBackground(Color.red);
269  }
270  }
271  catch(NumberFormatException e)
272  {
273  valide = false;
274  t_date.setBackground(Color.red);
275  }
276  }
277  }
278  if (c_lien.isSelected())
279  {
280  if ((!t_lien.getText().isEmpty()) && (!t_lien.getText().contains("https://www.youtube.com")))
281  {
282  valide = false;
283  t_lien.setBackground(Color.red);
284  }
285  }
286  if (c_lien_image.isSelected())
287  {
288  if (!t_lien_image.getText().isEmpty())
289  {
290  try
291  {
292  ImageIO.read(new URL(t_lien_image.getText()));
293  }
294  catch (IOException e)
295  {
296  valide = false;
297  t_lien_image.setBackground(Color.red);
298  }
299  }
300  }
301  return valide;
302  }

311  private void b_validerActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_b_validerActionPerformed
312  if(!champsvalides())
313  {
314  JOptionPane.showMessageDialog(this,"Des champs sont incorrects","Erreur",JOptionPane.ERROR_MESSAGE);
315  }
316  else
317  {
318  if (parent.getBDD().verifAuteurTitre(t_titre.getText(),t_auteur.getText(),parent.getPseudo()))
319  {
320  JOptionPane.showMessageDialog(this,"L'album existe déjà dans la base de données !","Erreur",JOptionPane.ERROR_MESSAGE);
321  }
322  else
323  {
324  if(c_titre.isSelected())
325  {
326  parent.modifier("titre",t_titre.getText(), panelparent.getDonnees().getTitre(), panelparent.getDonnees().getAuteur());
327  panelparent.getDonnees().setTitre(t_titre.getText());
328  panelparent.setTitre(panelparent.getDonnees().getTitre());
329  }
330  if(c_auteur.isSelected())
331  {
332  parent.modifier("auteur",t_auteur.getText(), panelparent.getDonnees().getTitre(), panelparent.getDonnees().getAuteur());
333  panelparent.getDonnees().setAuteur(t_auteur.getText());
334  panelparent.setAuteur(panelparent.getDonnees().getAuteur());
335  }
336  }
337 
338  if(c_date.isSelected())
339  {
340  parent.modifier("date",t_date.getText(), panelparent.getDonnees().getTitre(), panelparent.getDonnees().getAuteur());
341  panelparent.getDonnees().setDate(Integer.parseInt(t_date.getText()));
342  panelparent.setDate(panelparent.getDonnees().getDate());
343  }
344 
345  if(c_genre.isSelected())
346  {
347  parent.modifier("genre",b_genre.getSelectedItem().toString(), panelparent.getDonnees().getTitre(), panelparent.getDonnees().getAuteur());
348  panelparent.getDonnees().setGenre(b_genre.getSelectedItem().toString());
349  panelparent.setGenre(panelparent.getDonnees().getGenre());
350  }
351  if(c_lien.isSelected())
352  {
353  parent.modifier("lien",t_lien.getText(), panelparent.getDonnees().getTitre(), panelparent.getDonnees().getAuteur());
354  panelparent.getDonnees().setLien(t_lien.getText());
355  panelparent.setLien(panelparent.getDonnees().getLien());
356  }
357  if(c_lien_image.isSelected())
358  {
359  parent.modifier("lien_image",t_lien_image.getText(), panelparent.getDonnees().getTitre(), panelparent.getDonnees().getAuteur());
360  panelparent.getDonnees().setLienImage(t_lien_image.getText());
361  panelparent.setLienImage(panelparent.getDonnees().getLienImage());
362  }
363  parent.consulter();
364  dispose();
365  }
