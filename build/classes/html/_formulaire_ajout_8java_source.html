54  @Override
55  public boolean champsvalides()
56  {
57  boolean valide = true;
58  if (t_titre.getText().isEmpty())
59  {
60  valide = false;
61  t_titre.setBackground(Color.red);
62  }
63  if (t_auteur.getText().isEmpty())
64  {
65  valide = false;
66  t_auteur.setBackground(Color.red);
67  }
68  if (t_date.getText().isEmpty())
69  {
70  valide = false;
71  t_date.setBackground(Color.red);
72  }
73  else
74  {
75  try
76  {
77  if ((Integer.parseInt(t_date.getText()) > Integer.parseInt(this.dateactuelle) || (Integer.parseInt(t_date.getText()) < 0)))
78  {
79  valide = false;
80  t_date.setBackground(Color.red);
81  }
82  }
83  catch(NumberFormatException e)
84  {
85  valide = false;
86  t_date.setBackground(Color.red);
87  }
88  }
89  if ((!t_lien.getText().isEmpty()) && (!t_lien.getText().contains("https://www.youtube.com")))
90  {
91  valide = false;
92  t_lien.setBackground(Color.red);
93  }
94  if (!t_lien_image.getText().isEmpty())
95  {
96  try
97  {
98  ImageIO.read(new URL(t_lien_image.getText()));
99  }
100  catch (IOException e)
101  {
102  valide = false;
103  t_lien_image.setBackground(Color.red);
104  }
105  }
106  return valide;
107  }
